188def bezier(t, points):
189 n = len(points) - 1
190 x = sum(
191 binomial(n, i) * (1 - t) ** (n - i) * (t ** i) * points[i][0]
192 for i in range(n + 1)
193 )
194 y = sum(
195 binomial(n, i) * (1 - t) ** (n - i) * (t ** i) * points[i][1]
196 for i in range(n + 1)
197 )
198 return (x, y)

200def binomial(n, k):
201 if k < 0 or k > n:
202 return 0
203 if k == 0 or k == n:
204 return 1
205 k = min(k, n - k)
206 c = 1
207 for i in range(k):
208 c = c * (n - i) // (i + 1)
209 return c
