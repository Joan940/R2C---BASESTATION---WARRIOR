148def calculate_ball_angle(robot_x, robot_y, robot_angle, ball_x, ball_y):
149 # Menghitung delta X dan delta Y antara robot dan bola
150 dx = ball_x - robot_x
151 dy = ball_y - robot_y
152
153 # Menghitung sudut bola relatif terhadap robot
154 angle_to_ball = math.degrees(math.atan2(dy, dx))
155
156 # Menyesuaikan sudut dengan arah robot (kompas/sensor)
157 relative_angle = (angle_to_ball - robot_angle) % 360
158
159 if relative_angle < 0:
160 relative_angle += 360
161
162 return relative_angle

194def calculate_position(robot_x1, robot_y1, robot_angle1, robot_x2, robot_y2, robot_angle2):
195 # Mencari persamaan garis dari setiap robot
196 m1 = math.tan(math.radians(robot_angle1))
197 b1 = robot_y1 - m1 * robot_x1
198
199 m2 = math.tan(math.radians(robot_angle2))
200 b2 = robot_y2 - m2 * robot_x2
201
202 if m1 != m2:
203 intersection_x = (b2 - b1) / (m1 - m2)
204 intersection_y = m1 * intersection_x + b1
205 else:
206 intersection_x = -100
207 intersection_y = -100
208
209 return round(intersection_x), round(intersection_y)

402def bezier(t, points):
403 n = len(points) - 1
404 x = sum(
405 binomial(n, i) * (1 - t) ** (n - i) * (t ** i) * points[i][0]
406 for i in range(n + 1)
407 )
408 y = sum(
409 binomial(n, i) * (1 - t) ** (n - i) * (t ** i) * points[i][1]
410 for i in range(n + 1)
411 )
412 return (x, y)

421def binomial(n, k):
422 if k < 0 or k > n:
423 return 0
424 if k == 0 or k == n:
425 return 1
426 k = min(k, n - k)
427 c = 1
428 for i in range(k):
429 c = c * (n - i) // (i + 1)
430 return c
